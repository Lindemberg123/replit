Exemplos de Integração

JavaScript (Fetch)

fetch('http://your-repl-url/api/send-email', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer gmail-api-pro-key-2024'
  },
  body: JSON.stringify({
    to: 'destinatario@email.com',
    subject: 'Assunto do Email',
    message: 'Conteúdo da mensagem'
  })
})

cURL

curl -X POST http://your-repl-url/api/send-email \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer gmail-api-pro-key-2024" \
  -d '{"to":"destinatario@email.com","subject":"Teste","message":"Mensagem"}'
